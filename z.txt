const ProjectCard = ({ project }) => {

  const handlePurchasePackage = () => {
    // For existing projects that need package purchase

  };

  const handlePurchaseAddons = () => {
    // For adding addons to existing projects
    router.push(`/packages?type=addons-only&projectId=${project.publicId}`);
  };

  return (
    <Card>
      <CardContent>
        <h3>{project.title}</h3>
        <p>{project.projectType} ‚Ä¢ {project.areaSqFt} sq ft</p>

            </Button>
          )}
          <Button variant="outline" onClick={handlePurchaseAddons}>
            Add Add-ons
          </Button>
        </div>
      </CardContent>

  );
};
